<script>

  const game = document.getElementById('game');

  const player = document.getElementById('player');

  const scoreDisplay = document.getElementById('score');

  const message = document.getElementById('message');

  const jumpButton = document.getElementById('jumpButton');


  let score = 0;

  let jumping = false;

  let playerBottom = 20;

  const jumpHeight = 120;

  const gravity = 5;

  let jumpInterval;

  let fallInterval;


  const items = [];

  let gameInterval;

  let createItemInterval;


  function jump() {

    if (jumping) return;

    jumping = true;

    let upCount = 0;

    clearInterval(fallInterval);

    jumpInterval = setInterval(() => {

      if (upCount >= jumpHeight) {

        clearInterval(jumpInterval);

        fall();

      } else {

        playerBottom += gravity;

        player.style.bottom = playerBottom + 'px';

        upCount += gravity;

      }

    }, 20);

  }


  function fall() {

    fallInterval = setInterval(() => {

      if (playerBottom <= 20) {

        playerBottom = 20;

        player.style.bottom = playerBottom + 'px';

        jumping = false;

        clearInterval(fallInterval);

      } else {

        playerBottom -= gravity;

        player.style.bottom = playerBottom + 'px';

      }

    }, 20);

  }


  function createItem() {

    const item = document.createElement('div');

    item.classList.add('item');


    // 70% chance fruit, 30% sugar

    if (Math.random() < 0.7) {

      item.classList.add('fruit');

      item.type = 'fruit';

    } else {

      item.classList.add('sugar');

      item.type = 'sugar';

    }


    item.style.left = game.offsetWidth + 'px';

    item.style.top = (Math.random() * (game.offsetHeight - 80) + 40) + 'px';


    game.appendChild(item);

    items.push(item);

  }


  function moveItems() {

    for (let i = items.length - 1; i >= 0; i--) {

      const item = items[i];

      const left = parseInt(item.style.left);

      if (left < -50) {

        item.remove();

        items.splice(i, 1);

        continue;

      }

      item.style.left = (left - 5) + 'px';


      // Collision detection

      if (checkCollision(player, item)) {

        if (item.type === 'fruit') {

          score += 10;

          scoreDisplay.textContent = 'Score : ' + score;

          message.textContent = 'Super, un fruit attrapé ! 🍎';

        } else {

          score -= 15;

          scoreDisplay.textContent = 'Score : ' + score;

          message.textContent = 'Oups, sucre raffiné attrapé... 🍬';

        }

        item.remove();

        items.splice(i, 1);

      }

    }

  }


  function checkCollision(a, b) {

    const aRect = a.getBoundingClientRect();

    const bRect = b.getBoundingClientRect();


    return !(

      aRect.top > bRect.bottom ||

      aRect.bottom < bRect.top ||

      aRect.right < bRect.left ||

      aRect.left > bRect.right

    );

  }


  function startGame() {

    score = 0;

    scoreDisplay.textContent = 'Score : 0';

    message.textContent = '';

    items.forEach(item => item.remove());

    items.length = 0;

    playerBottom = 20;

    player.style.bottom = playerBottom + 'px';


    clearInterval(gameInterval);

    clearInterval(createItemInterval);


    createItemInterval = setInterval(createItem, 1500);

    gameInterval = setInterval(() => {

      moveItems();

    }, 30);


    setTimeout(() => {

      clearInterval(createItemInterval);

      clearInterval(gameInterval);

      message.textContent = `Temps écoulé ! Ton score final est ${score}. ${score >= 50 ? "Bravo, continue comme ça !" : "Essaie encore !"}`;

    }, 30000); // 30 secondes

  }


  jumpButton.addEventListener('click', jump);

  document.addEventListener('keydown', e => {

    if (e.code === 'Space') {

      jump();

    }

  });


  startGame();

</script>
